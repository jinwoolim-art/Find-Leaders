3.2 New normalizer

// src/modules/nec-importer/normalize-nec-location.ts
export async function normalizeNecCandidateLocation(
  candidate: NecCandidate,
  province: Province,
  deps: { findDistrict: (q: DistrictQuery) => Promise<ElectoralDistrict | null>;
          findSigungu: (provinceId: string, nameKo: string) => Promise<Sigungu | null>; }
): Promise<NormalizedLocation> {

  const typecode = candidate.sgTypecode;

  // 3 (governor) and 9 (superintendent) — province-only
  if (typecode === '3' || typecode === '9') {
    return { provinceId: province.id, sigunguId: null, electoralDistrictId: null };
  }

  // 4 (mayor/governor of sigungu) — sigungu-level, optional district
  if (typecode === '4') {
    const sigunguName = candidate.sggName ?? candidate.wiwName;
    const sigungu = sigunguName && sigunguName !== province.nameKo
      ? await deps.findSigungu(province.id, sigunguName)
      : null;
    const district = candidate.sggName
      ? await deps.findDistrict({ sgTypecode: '4', provinceId: province.id, sggName: candidate.sggName })
      : null;
    return { provinceId: province.id, sigunguId: sigungu?.id ?? null, electoralDistrictId: district?.id ?? null };
  }

  // 5, 7 (council) — district-level, sigungu derived from district
  if (typecode === '5' || typecode === '7') {
    const district = candidate.sggName
      ? await deps.findDistrict({ sgTypecode: typecode, provinceId: province.id, sggName: candidate.sggName })
      : null;
    const sigungu = district?.sigunguId
      ? { id: district.sigunguId }
      : (candidate.wiwName ? await deps.findSigungu(province.id, candidate.wiwName) : null);
    return { provinceId: province.id, sigunguId: sigungu?.id ?? null, electoralDistrictId: district?.id ?? null };
  }

  return { provinceId: province.id, sigunguId: null, electoralDistrictId: null };
}
findDistrict looks up ElectoralDistrict rows seeded from Excel (§5.1).

4. Filter Plan — one shared strict matcher
4.1 Extract a single source of truth

// src/modules/candidate-filter/public-candidate-filter.ts
export type PublicListFilter = {
  electionTypeId: string;
  provinceId: string;
  sigunguId?: string;
  emdId?: string;
};

export type CandidateProfileForFilter = {
  electionTypeId: string;
  provinceId: string | null;
  sigunguId: string | null;
  electoralDistrictId: string | null;
};

/**
 * Strict rules:
 *  - electionType must match exactly
 *  - province must match exactly
 *  - sigunguId filter requires profile.sigunguId === filter.sigunguId (no null fallback)
 *  - emdId filter:
 *      profile.electoralDistrictId must be present AND
 *      district must cover the chosen emd
 *      (NO fallback to sigungu or province)
 */
export async function profileMatchesPublicFilter(
  profile: CandidateProfileForFilter,
  filter: PublicListFilter,
  deps: { districtCoversEmd: (districtId: string, emdId: string) => Promise<boolean> }
): Promise<boolean> {
  if (profile.electionTypeId !== filter.electionTypeId) return false;
  if (profile.provinceId !== filter.provinceId) return false;

  if (filter.emdId) {
    if (!profile.electoralDistrictId) return false;
    return await deps.districtCoversEmd(profile.electoralDistrictId, filter.emdId);
  }
  if (filter.sigunguId) {
    return profile.sigunguId === filter.sigunguId;
  }
  return true;
}
